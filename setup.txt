NOTE: This is all done on a Ubuntu WSL fresh install, just download the repo and cd into it, then run the commands

# Install Packages
sudo apt update && sudo apt upgrade
sudo apt install mysql-server python3 python3-pip python3-venv
# Setup MySQL
sudo mysql -u root
# Edit values in quotations to whatever you want, just adjust the .env accordingly
CREATE DATABASE socketfit;
# Note, i had to use root cuz privileges were being weird with other users, look into later
ALTER USER 'root'@'localhost' IDENTIFIED WITH caching_sha2_password BY 'password';
FLUSH PRIVILEGES;
# Setup Python Environment
python3 -m venv .venv
source .venv/bin/activate
# Install requirements
pip install -r ./requirements.txt
# Setup database
python ./src/setup.py
# Run program
fastapi dev ./src/main.py


DOCKER BUILDING:
sudo docker build -t socketfit .
sudo docker compose up
